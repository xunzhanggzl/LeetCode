class Solution(object):
    def maxSubArray(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        dp = copy.copy(nums)
        for i in range(1, len(nums)):
            dp[i] = max(dp[i], dp[i] + dp[i-1])
        return max(dp)
