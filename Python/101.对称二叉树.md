# [101. 对称二叉树](https://leetcode-cn.com/problems/symmetric-tree/)

## 1. 递归

```python
# Definition for a binary tree node.
# class TreeNode(object):
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution(object):
    def isSymmetric(self, root):
        """
        :type root: TreeNode
        :rtype: bool
        """
        if not root:
            return True
        return self.dfs(root.left, root.right)
    def dfs(self, l, r):
        if (not l and not r):
            return True
        if (l and not r) or (not l and r) or (l.val != r.val):
            return False
        return self.dfs(l.left, r.right) and self.dfs(l.right, r.left)

```

## 2. 迭代

python中不知道有没有像JavaScript中的shift那样的方法，这里用了pop(0) 和 队列的原理

```python
# Definition for a binary tree node.
# class TreeNode(object):
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution(object):
    def isSymmetric(self, root):
        if not root:
            return True
        q1 = [root.left]
        q2 = [root.right]
        while len(q1) or len(q2):
            n1 = q1.pop(0)
            n2 = q2.pop(0)
            if not n1 and not n2:
                continue
            if (not n1 and n2) or (n1 and not n2) or (n1.val != n2.val):
                return False
            q1.append(n1.left)
            q1.append(n1.right)
            q2.append(n2.right)
            q2.append(n2.left)
        return True
```
