# [46. 全排列](https://leetcode-cn.com/problems/permutations/)

回溯思想 + 函数传参

```javascript
/**
 * @param {number[]} nums
 * @return {number[][]}
 */
var permute = function (nums) {
    let res = [];
    function backTracking(temp, remain) {
        if (remain.length === 0) {
            res.push(temp.slice());
        }
        for(let i = 0; i < remain.length; i ++) {
            temp.push(remain[i]);
            backTracking(temp, remain.slice(0,i).concat(remain.slice(i+1)));
            temp.pop();
        }
    }

    backTracking([], nums)
    return res;
};
```
