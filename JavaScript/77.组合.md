# [组合](https://leetcode-cn.com/problems/combinations/)

## 1. 回溯

```javascript
/**
 * @param {number} n
 * @param {number} k
 * @return {number[][]}
 */
var combine = function (n, k) {
    let res = [];
    let arr = [];
    for(let i = 1; i <= n; i ++) {
        arr.push(i);
    }
    function backTracking(temp, remain) {
        if (temp.slice().length === k) {
            res.push(temp.slice());
            return;
        }
        for(let i = 0; i < remain.length; i ++) {
            temp.push(remain[i]);
            backTracking(temp, remain.slice(i+1));
            temp.pop();
        }
    }
    backTracking([], arr);

    return res;
};
```
